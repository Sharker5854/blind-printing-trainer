CREATE TABLE "Users" ("id" INTEGER NOT NULL UNIQUE,"username" TEXT NOT NULL UNIQUE, PRIMARY KEY("id" AUTOINCREMENT))

CREATE TABLE "Statistics" ("id" INTEGER NOT NULL UNIQUE, "user_id" INTEGER NOT NULL UNIQUE, "total_rounds" INTEGER NOT NULL DEFAULT 0, "total_time_minutes" REAL NOT NULL DEFAULT 0, "total_symbols" INTEGER NOT NULL DEFAULT 0, "average_symbols_in_minute" INTEGER NOT NULL DEFAULT 0, "total_mistakes" INTEGER NOT NULL DEFAULT 0, "average_mistakes_per_round" INTEGER NOT NULL DEFAULT 0, PRIMARY KEY("id" AUTOINCREMENT), FOREIGN KEY("user_id") REFERENCES Users(id))
